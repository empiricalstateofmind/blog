class Timer(object):
    
    def __init__(self, num_samples=None):
        self.num_samples = num_samples
    
    def __enter__(self):
        self.start = time.clock()
        return self
    
    def __exit__(self, *args, **kwargs):
        self.end = time.clock()
        print("The functions took {} seconds,".format(self.end-self.start))
        if self.num_samples is not None:
            print("equating to {} seconds per run.".format((self.end-self.start)/self.num_samples))

Stochastic Differential Equations¶

Reference Gardiner

$$ dX_t = f(X,t)dt + g(X,t) dW_t $$$$ X(t+\Delta t) = X(t) + f(X,t) dt + g(X,t) dW $$$$ X(t+\Delta t) = X(t) + \mu X dt + \sigma X dW $$

# Drift
def f(x, mu):
    return mu*x

# Volatility
def g(x, sigma):
    return sigma*x

# Stepping Function
def python_euler_step(x, mu, sigma, dt):
    result = x + dt*f(x, mu) + np.sqrt(dt)*g(x, sigma)*np.random.randn()
    result = 2000 - result if result > 1000 else result
    return result

dt0 = 1
total_time = 1000
total_steps = int(total_time/dt0)
x0 = 100
mu0 = 0.0
sigma0 = 0.1

with Timer():
    xs = np.empty(shape=total_steps, dtype=float)
    xs[0] = x0
    for step in range(1,total_steps):
        xs[step] = python_euler_step(xs[step-1], mu0, sigma0, dt0)
    
plt.plot(xs)

# Drift
def f(x, mu):
    return mu*x

# Volatility
def g(x, sigma):
    return sigma*x

# Stepping Function
def euler_step(x, mu, sigma, dt):
    result = x + dt*f(x, mu) + T.sqrt(dt)*g(x, sigma)*rv_n
    #if T.gt(result, 200):
    #    result = 400 - result
    result = T.switch(T.gt(result, 1000), T.cast(2000 - result, 'float32'), T.cast(result, 'float32'))    
    return result
    #return T.cast(result, 'float32')

# Random Number Generator
srng = RandomStreams(seed=12345)

# Symbolic Variables
dt = T.fscalar("dt")
mu = T.fscalar("mu")
sigma = T.fscalar("sigma")
x = T.fvector("x")

# Numeric Variables
num_samples = 5000
dt0 = 1 # Timestep
total_time = 1000
total_steps = int(total_time/dt0)

mu0 = 0.0
sigma0 = 0.1

x0 = theano.shared(100*np.ones(num_samples, dtype='float32')) # IC
rv_n = srng.normal(x.shape, std=1) # Gaussian Noise

# Iteration Loop
(xout, updates) = theano.scan(fn=euler_step,
                                outputs_info=[x], #output shape
                                non_sequences=[mu, sigma, dt], #fixed parameters
                                n_steps=total_steps)
#compile it
sim = theano.function(inputs=[mu, sigma, dt], 
                    outputs=xout, 
                    givens={x:x0}, 
                    updates=updates,
                    allow_input_downcast=True)

#print("Running sim...")
#start = time.clock()
with Timer(num_samples):
    xout = sim(mu0, sigma0, dt0)

# Drift
def f(x, k1, k2, k3, k4, k5):
    return -k1*T.pow(x,3) + k2*T.pow(x,2) - k3*x + k4

# Volatility
def g(x, k1, k2, k3, k4, k5):
    return k5

def euler_step(x, k1, k2, k3, k4, k5, dt):
    return T.cast(x + dt*f(x, k1, k2, k3, k4, k5) + T.sqrt(dt)*g(x, k1, k2, k3, k4, k5)*rv_n, 'float32')

# RNG
srng = RandomStreams(seed=12345)

# Symbolic variables
dt = T.fscalar("dt")
k1 = T.fscalar("k1")
k2 = T.fscalar("k2")
k3 = T.fscalar("k3")
k4 = T.fscalar("k4")
k5 = T.fscalar("k5")
x = T.fvector("x")

# Define numeric variables
num_samples = 5000
x0 = theano.shared(250*np.ones(num_samples, dtype='float32')) # IC

k10 = 1e-3
k20 = 0.75
k30 = 165
k40 = 1e4
k50 = 200

dt0 = 0.01 # Timestep
total_time = 1000
total_steps = int(total_time/dt0)
rv_n = srng.normal(x.shape, avg=0, std=1) # Gaussian Noise

# Iteration Loop
(xout, updates) = theano.scan(fn=euler_step,
                                outputs_info=[x], #output shape
                                non_sequences=[k1, k2, k3, k4, k5, dt], #fixed parameters
                                n_steps=total_steps)
#compile it
sim = theano.function(inputs=[k1, k2, k3, k4, k5, dt], 
                    outputs=xout, 
                    givens={x:x0}, 
                    updates=updates,
                    allow_input_downcast=True)

print("Running sim...")
start = time.clock()
xout = sim(k10, k20, k30, k40, k50, dt0)
diff = (time.clock() - start)
print("done in", diff, "s at ", diff/num_samples, "s per path")
